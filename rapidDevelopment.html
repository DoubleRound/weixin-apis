最简单的一个示例

var express = require('express');
var weixin = require('weixin-apis');
var app = express();

weixin.configurate({
	app: app,
	token : '你的token',
	appid : '你的appid（可选）',
	secret : '你的secret（可选）'
});

weixin.on('textMsg', function(data) {
	var msg = {
		toUserName : data.fromUserName,
		fromUserName : data.toUserName,
		msgType : 'text',
		content : data.content
	};
	weixin.sendMsg(msg);
})

app.listen(80);
